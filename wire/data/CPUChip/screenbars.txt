//ATTACH TO DIGITAL SCREEN:
//PORT0 - PIXELX
//PORT1 - PIXELY
//PORT2 - PIXELG
DATA;
alloc color;
CODE;
mov #color,128;//Color will be from 128 to 255!
mov eax,0; //EAX will be X, and EBX will be Y
loopx:
  mov port0,eax;
  mov ebx,0;
  loopy:
    mov port1,ebx;
    add #color,8;
    cmp #color,255;
    jle if1_false;
      mov #color,128;
    if1_false:
    mov port2,#color;
    inc ebx;
    cmp ebx,32;
    jl loopy;
  inc eax;
  cmp eax,32;
  jl loopx;
